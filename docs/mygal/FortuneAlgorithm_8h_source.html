93  bool bound(Box<T> box)
94  {
95  // Make sure the bounding box contains all the vertices
96  for (const auto& vertex : mDiagram.getVertices()) // Much faster when using vector<unique_ptr<Vertex*>, maybe we can test vertices in border cells to speed up
97  {
98  box.left = std::min(vertex.point.x, box.left);
99  box.bottom = std::min(vertex.point.y, box.bottom);
100  box.right = std::max(vertex.point.x, box.right);
101  box.top = std::max(vertex.point.y, box.top);
102  }
103  // Retrieve all non bounded half edges from the beach line
104  auto linkedVertices = std::list<LinkedVertex>();
105  auto vertices = std::unordered_map<std::size_t, std::array<LinkedVertex*, 8>>(mDiagram.getNbSites());
106  if (!mBeachline.isEmpty())
107  {
108  auto leftArc = mBeachline.getLeftmostArc();
109  auto rightArc = leftArc->next;
110  while (!mBeachline.isNil(rightArc))
111  {
112  // Bound the edge
113  auto direction = (leftArc->site->point - rightArc->site->point).getOrthogonal();
114  auto origin = (leftArc->site->point + rightArc->site->point) * static_cast<T>(0.5);
115  // Line-box intersection
116  auto intersection = box.getFirstIntersection(origin, direction);
117  // Create a new vertex and ends the half edges
118  auto vertex = mDiagram.createVertex(intersection.point);
119  setDestination(leftArc, rightArc, vertex);
120  // Initialize pointers
121  if (vertices.find(leftArc->site->index) == vertices.end())
122  vertices[leftArc->site->index].fill(nullptr);
123  if (vertices.find(rightArc->site->index) == vertices.end())
124  vertices[rightArc->site->index].fill(nullptr);
125  // Store the vertex on the boundaries
126  if (vertices[leftArc->site->index][2 * static_cast<int>(intersection.side) + 1])
127  std::cout << "error" << std::endl;
128  linkedVertices.emplace_back(LinkedVertex{nullptr, vertex, leftArc->rightHalfEdge});
129  vertices[leftArc->site->index][2 * static_cast<int>(intersection.side) + 1] = &linkedVertices.back();
130  if (vertices[rightArc->site->index][2 * static_cast<int>(intersection.side)])
131  std::cout << "error" << std::endl;
132  linkedVertices.emplace_back(LinkedVertex{rightArc->leftHalfEdge, vertex, nullptr});
133  vertices[rightArc->site->index][2 * static_cast<int>(intersection.side)] = &linkedVertices.back();
134  // Next edge
135  leftArc = rightArc;
136  rightArc = rightArc->next;
137  }
138  }
139  // Add corners
140  for (auto& kv : vertices)
141  {
142  auto& cellVertices = kv.second;
143  // We check twice the first side to be sure that all necessary corners are added
144  for (std::size_t i = 0; i < 5; ++i)
145  {
146  auto side = i % 4;
147  auto nextSide = (side + 1) % 4;
148  // Add first corner
149  if (cellVertices[2 * side] == nullptr && cellVertices[2 * side + 1] != nullptr)
150  {
151  auto prevSide = (side + 3) % 4;
152  auto corner = mDiagram.createCorner(box, static_cast<typename Box<T>::Side>(side));
153  linkedVertices.emplace_back(LinkedVertex{nullptr, corner, nullptr});
154  cellVertices[2 * prevSide + 1] = &linkedVertices.back();
155  cellVertices[2 * side] = &linkedVertices.back();
156  }
157  // Add second corner
158  else if (cellVertices[2 * side] != nullptr && cellVertices[2 * side + 1] == nullptr)
159  {
160  auto corner = mDiagram.createCorner(box, static_cast<typename Box<T>::Side>(nextSide));
161  linkedVertices.emplace_back(LinkedVertex{nullptr, corner, nullptr});
162  cellVertices[2 * side + 1] = &linkedVertices.back();
163  cellVertices[2 * nextSide] = &linkedVertices.back();
164  }
165  }
166  }
167  // Join the half edges
168  for (auto& kv : vertices)
169  {
170  auto i = kv.first;
171  auto& cellVertices = kv.second;
172  for (std::size_t side = 0; side < 4; ++side)
173  {
174  if (cellVertices[2 * side] != nullptr)
175  {
176  // Link vertices
177  auto halfEdge = mDiagram.createHalfEdge(mDiagram.getFace(i));
178  halfEdge->origin = cellVertices[2 * side]->vertex;
179  halfEdge->destination = cellVertices[2 * side + 1]->vertex;
180  cellVertices[2 * side]->nextHalfEdge = halfEdge;
181  halfEdge->prev = cellVertices[2 * side]->prevHalfEdge;
182  if (cellVertices[2 * side]->prevHalfEdge != nullptr)
183  cellVertices[2 * side]->prevHalfEdge->next = halfEdge;
184  cellVertices[2 * side + 1]->prevHalfEdge = halfEdge;
185  halfEdge->next = cellVertices[2 * side + 1]->nextHalfEdge;
186  if (cellVertices[2 * side + 1]->nextHalfEdge != nullptr)
187  cellVertices[2 * side + 1]->nextHalfEdge->prev = halfEdge;
188  }
189  }
190  }
191  return true; // TO DO: detect errors
192  }

214  void handleSiteEvent(Event<T>* event)
215  {
216  auto site = event->site;
217  // 1. Check if the bachline is empty
218  if (mBeachline.isEmpty())
219  {
220  mBeachline.setRoot(mBeachline.createArc(site));
221  return;
222  }
223  // 2. Look for the arc above the site
224  auto arcToBreak = mBeachline.locateArcAbove(site->point, mBeachlineY);
225  deleteEvent(arcToBreak);
226  // 3. Replace this arc by the new arcs
227  auto middleArc = breakArc(arcToBreak, site);
228  auto leftArc = middleArc->prev;
229  auto rightArc = middleArc->next;
230  // 4. Add an edge in the diagram
231  addEdge(leftArc, middleArc);
232  middleArc->rightHalfEdge = middleArc->leftHalfEdge;
233  rightArc->leftHalfEdge = leftArc->rightHalfEdge;
234  // 5. Check circle events
235  // Left triplet
236  if (!mBeachline.isNil(leftArc->prev))
237  addEvent(leftArc->prev, leftArc, middleArc);
238  // Right triplet
239  if (!mBeachline.isNil(rightArc->next))
240  addEvent(middleArc, rightArc, rightArc->next);
241  }

243  void handleCircleEvent(Event<T>* event)
244  {
245  auto point = event->point;
246  auto arc = event->arc;
247  // 1. Add vertex
248  auto vertex = mDiagram.createVertex(point);
249  // 2. Delete all the events with this arc
250  auto leftArc = arc->prev;
251  auto rightArc = arc->next;
252  deleteEvent(leftArc);
253  deleteEvent(rightArc);
254  // 3. Update the beachline and the diagram
255  removeArc(arc, vertex);
256  // 4. Add new circle events
257  // Left triplet
258  if (!mBeachline.isNil(leftArc->prev))
259  addEvent(leftArc->prev, leftArc, rightArc);
260  // Right triplet
261  if (!mBeachline.isNil(rightArc->next))
262  addEvent(leftArc, rightArc, rightArc->next);
263  }

267  Arc<T>* breakArc(Arc<T>* arc, typename Diagram<T>::Site* site)
268  {
269  // Create the new subtree
270  auto middleArc = mBeachline.createArc(site);
271  auto leftArc = mBeachline.createArc(arc->site);
272  leftArc->leftHalfEdge = arc->leftHalfEdge;
273  auto rightArc = mBeachline.createArc(arc->site);
274  rightArc->rightHalfEdge = arc->rightHalfEdge;
275  // Insert the subtree in the beachline
276  mBeachline.replace(arc, middleArc);
277  mBeachline.insertBefore(middleArc, leftArc);
278  mBeachline.insertAfter(middleArc, rightArc);
279  // Delete old arc
280  delete arc;
281  // Return the middle arc
282  return middleArc;
283  }

285  void removeArc(Arc<T>* arc, typename Diagram<T>::Vertex* vertex)
286  {
287  // End edges
288  setDestination(arc->prev, arc, vertex);
289  setDestination(arc, arc->next, vertex);
290  // Join the edges of the middle arc
291  arc->leftHalfEdge->next = arc->rightHalfEdge;
292  arc->rightHalfEdge->prev = arc->leftHalfEdge;
293  // Update beachline
294  mBeachline.remove(arc);
295  // Create a new edge
296  auto prevHalfEdge = arc->prev->rightHalfEdge;
297  auto nextHalfEdge = arc->next->leftHalfEdge;
298  addEdge(arc->prev, arc->next);
299  setOrigin(arc->prev, arc->next, vertex);
300  setPrevHalfEdge(arc->prev->rightHalfEdge, prevHalfEdge);
301  setPrevHalfEdge(nextHalfEdge, arc->next->leftHalfEdge);
302  // Delete node
303  delete arc;
304  }

350  void addEvent(Arc<T>* left, Arc<T>* middle, Arc<T>* right)
351  {
352  auto y = T();
353  auto convergencePoint = computeConvergencePoint(left->site->point, middle->site->point, right->site->point, y);
354  auto isBelow = y <= mBeachlineY;
355  auto leftBreakpointMovingRight = isMovingRight(left, middle);
356  auto rightBreakpointMovingRight = isMovingRight(middle, right);
357  auto leftInitialX = getInitialX(left, middle, leftBreakpointMovingRight);
358  auto rightInitialX = getInitialX(middle, right, rightBreakpointMovingRight);
359  auto isValid =
360  ((leftBreakpointMovingRight && leftInitialX < convergencePoint.x) ||
361  (!leftBreakpointMovingRight && leftInitialX > convergencePoint.x)) &&
362  ((rightBreakpointMovingRight && rightInitialX < convergencePoint.x) ||
363  (!rightBreakpointMovingRight && rightInitialX > convergencePoint.x));
364  if (isValid && isBelow)
365  {
366  auto event = std::make_unique<Event<T>>(y, convergencePoint, middle);
367  middle->event = event.get();
368  mEvents.push(std::move(event));
369  }
370  }

381  Vector2<T> computeConvergencePoint(const Vector2<T>& point1, const Vector2<T>& point2, const Vector2<T>& point3, T& y) const
382  {
383  auto v1 = (point1 - point2).getOrthogonal();
384  auto v2 = (point2 - point3).getOrthogonal();
385  auto delta = static_cast<T>(0.5) * (point3 - point1);
386  auto t = delta.getDet(v2) / v1.getDet(v2);
387  auto center = static_cast<T>(0.5) * (point1 + point2) + t * v1;
388  auto r = center.getDistance(point1);
389  y = center.y - r;
390  return center;
391  }
