182  bool intersect(Box<T> box)
183  {
184  auto error = false;
185  auto processedHalfEdges = std::unordered_set<HalfEdge*>();
186  auto verticesToRemove = std::unordered_set<Vertex*>();
187  for (const auto& site : mSites)
188  {
189  auto halfEdge = site.face->outerComponent;
190  auto inside = box.contains(halfEdge->origin->point);
191  auto outerComponentDirty = !inside;
192  auto incomingHalfEdge = static_cast<HalfEdge*>(nullptr); // First half edge coming in the box
193  auto outgoingHalfEdge = static_cast<HalfEdge*>(nullptr); // Last half edge going out the box
194  auto incomingSide = typename Box<T>::Side{};
195  auto outgoingSide = typename Box<T>::Side{};
196  do
197  {
198  auto intersections = std::array<typename Box<T>::Intersection, 2>{};
199  auto nbIntersections = box.getIntersections(halfEdge->origin->point, halfEdge->destination->point, intersections);
200  auto nextInside = box.contains(halfEdge->destination->point);
201  auto nextHalfEdge = halfEdge->next;
202  // The two points are outside the box
203  if (!inside && !nextInside)
204  {
205  // The edge is outside the box
206  if (nbIntersections == 0)
207  {
208  verticesToRemove.emplace(halfEdge->origin);
209  removeHalfEdge(halfEdge);
210  }
211  // The edge crosses twice the frontiers of the box
212  else if (nbIntersections == 2)
213  {
214  verticesToRemove.emplace(halfEdge->origin);
215  if (processedHalfEdges.find(halfEdge->twin) != processedHalfEdges.end())
216  {
217  halfEdge->origin = halfEdge->twin->destination;
218  halfEdge->destination = halfEdge->twin->origin;
219  }
220  else
221  {
222  halfEdge->origin = createVertex(intersections[0].point);
223  halfEdge->destination = createVertex(intersections[1].point);
224  }
225  if (outgoingHalfEdge != nullptr)
226  link(box, outgoingHalfEdge, outgoingSide, halfEdge, intersections[0].side);
227  if (incomingHalfEdge == nullptr)
228  {
229  incomingHalfEdge = halfEdge;
230  incomingSide = intersections[0].side;
231  }
232  outgoingHalfEdge = halfEdge;
233  outgoingSide = intersections[1].side;
234  processedHalfEdges.emplace(halfEdge);
235  }
236  else
237  error = true;
238  }
239  // The edge is going outside the box
240  else if (inside && !nextInside)
241  {
242  if (nbIntersections == 1)
243  {
244  if (processedHalfEdges.find(halfEdge->twin) != processedHalfEdges.end())
245  halfEdge->destination = halfEdge->twin->origin;
246  else
247  halfEdge->destination = createVertex(intersections[0].point);
248  outgoingHalfEdge = halfEdge;
249  outgoingSide = intersections[0].side;
250  processedHalfEdges.emplace(halfEdge);
251  }
252  else
253  error = true;
254  }
255  // The edge is coming inside the box
256  else if (!inside && nextInside)
257  {
258  if (nbIntersections == 1)
259  {
260  verticesToRemove.emplace(halfEdge->origin);
261  if (processedHalfEdges.find(halfEdge->twin) != processedHalfEdges.end())
262  halfEdge->origin = halfEdge->twin->destination;
263  else
264  halfEdge->origin = createVertex(intersections[0].point);
265  if (outgoingHalfEdge != nullptr)
266  link(box, outgoingHalfEdge, outgoingSide, halfEdge, intersections[0].side);
267  if (incomingHalfEdge == nullptr)
268  {
269  incomingHalfEdge = halfEdge;
270  incomingSide = intersections[0].side;
271  }
272  processedHalfEdges.emplace(halfEdge);
273  }
274  else
275  error = true;
276  }
277  halfEdge = nextHalfEdge;
278  // Update inside
279  inside = nextInside;
280  } while (halfEdge != site.face->outerComponent);
281  // Link the last and the first half edges inside the box
282  if (outerComponentDirty && incomingHalfEdge != nullptr)
283  link(box, outgoingHalfEdge, outgoingSide, incomingHalfEdge, incomingSide);
284  // Set outer component
285  if (outerComponentDirty)
286  site.face->outerComponent = incomingHalfEdge;
287  }
288  // Remove vertices
289  for (auto& vertex : verticesToRemove)
290  removeVertex(vertex);
291  // Return the status
292  return !error;
293  }

306  std::vector<Vector2<T>> computeLloydRelaxation() const
307  {
308  auto sites = std::vector<Vector2<T>>();
309  for (const auto& face : mFaces)
310  {
311  auto area = static_cast<T>(0.0);
312  auto centroid = Vector2<T>();
313  auto halfEdge = face.outerComponent;
314  do
315  {
316  auto det = halfEdge->origin->point.getDet(halfEdge->destination->point);
317  area += det;
318  centroid += (halfEdge->origin->point + halfEdge->destination->point) * det;
319  halfEdge = halfEdge->next;
320  } while (halfEdge != face.outerComponent);
321  area *= 0.5;
322  centroid *= 1.0 / (6.0 * area);
323  sites.push_back(centroid);
324  }
325  return sites;
326  }

340  Triangulation computeTriangulation() const
341  {
342  auto neighbors = std::vector<std::vector<std::size_t>>(mSites.size());
343  for (std::size_t i = 0; i < mSites.size(); ++i)
344  {
345  auto face = mFaces[i];
346  auto halfEdge = face.outerComponent;
347  while (halfEdge->prev != nullptr)
348  {
349  halfEdge = halfEdge->prev;
350  if(halfEdge == face.outerComponent)
351  break;
352  }
353  while (halfEdge != nullptr)
354  {
355  if (halfEdge->twin != nullptr)
356  neighbors[i].push_back(halfEdge->twin->incidentFace->site->index);
357  halfEdge = halfEdge->next;
358  if(halfEdge == face.outerComponent)
359  break;
360  }
361  }
362  return Triangulation(neighbors);
363  }

425  void link(Box<T> box, HalfEdge* start, typename Box<T>::Side startSide, HalfEdge* end, typename Box<T>::Side endSide)
426  {
427  auto halfEdge = start;
428  auto side = static_cast<int>(startSide);
429  while (side != static_cast<int>(endSide))
430  {
431  side = (side + 1) % 4;
432  halfEdge->next = createHalfEdge(start->incidentFace);
433  halfEdge->next->prev = halfEdge;
434  halfEdge->next->origin = halfEdge->destination;
435  halfEdge->next->destination = createCorner(box, static_cast<typename Box<T>::Side>(side));
436  halfEdge = halfEdge->next;
437  }
438  halfEdge->next = createHalfEdge(start->incidentFace);
439  halfEdge->next->prev = halfEdge;
440  end->prev = halfEdge->next;
441  halfEdge->next->next = end;
442  halfEdge->next->origin = halfEdge->destination;
443  halfEdge->next->destination = end->origin;
444  }
